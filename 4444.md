-- LocalScript: Teleporte/Tween para NPC mais próximo (toggle UI) - ANTI-SUMIR!
local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local rootPart = character:WaitForChild("HumanoidRootPart")

-- ✅ CORRIGIDO: workspace.Enemies
local ENEMIES_FOLDER = workspace:WaitForChild("Enemies")

print("✅ Teleporte carregado! Pasta:", ENEMIES_FOLDER.Name)

-- Configurações
local TWEEN_TIME = 0.1
local UPDATE_INTERVAL = 0.1
local MIN_DISTANCE_TO_TP = 8
local USE_TWEEN = false

-- Toggle global
local enabled = false

-- ✅ CORRIGIDO: Usa CoreGui (igual Mini Hub) - NUNCA SOME!
local function createToggleUI()
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "EnemyTeleportUI"
    screenGui.ResetOnSpawn = false
    screenGui.Parent = game:GetService("CoreGui")  -- ← AQUI É O SEGREDO!

    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 140, 0, 50)
    frame.Position = UDim2.new(1, -150, 0, 20)
    frame.BackgroundColor3 = Color3.fromRGB(30, 30, 40)
    frame.BorderSizePixel = 0
    frame.Parent = screenGui

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = frame

    local toggleButton = Instance.new("TextButton")
    toggleButton.Size = UDim2.new(1, -10, 1, -10)
    toggleButton.Position = UDim2.new(0, 5, 0, 5)
    toggleButton.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
    toggleButton.TextColor3 = Color3.new(1,1,1)
    toggleButton.Font = Enum.Font.GothamBold
    toggleButton.TextSize = 16
    toggleButton.Text = "Teleport: OFF"
    toggleButton.Parent = frame

    local uiCornerBtn = Instance.new("UICorner")
    uiCornerBtn.CornerRadius = UDim.new(0, 6)
    uiCornerBtn.Parent = toggleButton

    toggleButton.MouseButton1Click:Connect(function()
        enabled = not enabled
        
        if enabled then
            toggleButton.Text = "Teleport: ON"
            toggleButton.BackgroundColor3 = Color3.fromRGB(0, 170, 100)
            print("🔥 TELEPORTE ATIVADO!")
        else
            toggleButton.Text = "Teleport: OFF"
            toggleButton.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
            print("❌ Teleporte DESATIVADO")
        end
    end)

    return toggleButton
end

createToggleUI()

-- Função: acha o inimigo mais próximo
local function getClosestEnemy()
    local closest = nil
    local smallestDist = math.huge

    for _, model in ipairs(ENEMIES_FOLDER:GetChildren()) do
        if not model:IsA("Model") then continue end
        
        local hrp = model:FindFirstChild("HumanoidRootPart") 
                 or model:FindFirstChild("Torso") 
                 or model:FindFirstChild("UpperTorso")
                 or model.PrimaryPart
        
        if not hrp then continue end

        local dist = (hrp.Position - rootPart.Position).Magnitude
        if dist < smallestDist then
            smallestDist = dist
            closest = hrp
        end
    end

    return closest, smallestDist
end

-- Tween suave
local function tweenToTarget(targetPart)
    if not targetPart then return end
    local offset = (rootPart.Position - targetPart.Position).Unit * 4
    local desiredPos = targetPart.Position + offset
    local goalCFrame = CFrame.new(desiredPos, targetPart.Position)
    
    local tweenInfo = TweenInfo.new(TWEEN_TIME, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
    local tween = TweenService:Create(rootPart, tweenInfo, {CFrame = goalCFrame})
    tween:Play()
end

-- TP instantâneo
local function instantTP(targetPart)
    if not targetPart then return end
    local offset = (rootPart.Position - targetPart.Position).Unit * 4
    local desiredPos = targetPart.Position + offset
    rootPart.CFrame = CFrame.new(desiredPos, targetPart.Position)
end

-- Loop principal
local lastUpdate = 0
local tpCount = 0

RunService.Heartbeat:Connect(function()
    if not enabled then return end
    if tick() - lastUpdate < UPDATE_INTERVAL then return end
    
    lastUpdate = tick()
    
    local closestPart, distance = getClosestEnemy()
    
    if not closestPart then return end
    if distance < MIN_DISTANCE_TO_TP then return end
    
    tpCount = tpCount + 1
    print("⚡ TP #"..tpCount.." → "..math.floor(distance).." studs")
    
    if USE_TWEEN then
        tweenToTarget(closestPart)
    else
        instantTP(closestPart)
    end
end)

-- Recarrega character
player.CharacterAdded:Connect(function(newChar)
    character = newChar
    rootPart = newChar:WaitForChild("HumanoidRootPart")
    print("👤 Character reload")
end)

print("🎉 TELEPORTE PRONTO! CoreGui = NUNCA SOME!")
