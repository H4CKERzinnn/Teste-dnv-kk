-- LocalScript: AUTO PROXIMITY PROMPT (SEMPRE ATIVO) - Rewards + TODOS
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local rootPart = character:WaitForChild("HumanoidRootPart")

print("✅ Auto Proximity SEMPRE ATIVO!")

-- 🔥 CONFIGS
local PROXIMITY_RANGE = 15  -- Studs
local UPDATE_INTERVAL = 0.1
local DEBOUNCE_TIME = 0.5

-- Controle
local lastUpdate = 0
local triggeredPrompts = {}

-- 🔥 AUTO PROXIMITY (SEM UI - SEMPRE LIGADO)
local function findAndTriggerPrompts()
    local playerPos = rootPart.Position
    
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if not obj:IsA("ProximityPrompt") then continue end
        
        -- FILTRO REWARDS (opcional - mude pra true se quiser só rewards)
        local onlyRewards = false
        if onlyRewards then
            local isReward = string.find(string.lower(obj.Parent.Name), "reward") 
                          or string.find(string.lower(obj.Parent.Parent.Name), "reward")
                          or string.find(string.lower(obj.Name), "reward")
            if not isReward then continue end
        end
        
        -- DISTÂNCIA
        local promptHolder = obj.Parent
        local promptPos = promptHolder:IsA("BasePart") and promptHolder.Position 
                       or promptHolder:FindFirstChild("HumanoidRootPart") 
                       or promptHolder.PrimaryPart
                       or promptHolder:GetPivot().Position
        
        local distance = (Vector3.new(promptPos.X, playerPos.Y, promptPos.Z) - playerPos).Magnitude
        
        if distance > PROXIMITY_RANGE then continue end
        
        -- ANTI-SPAM
        local promptKey = obj:GetDebugId()
        if triggeredPrompts[promptKey] and tick() - triggeredPrompts[promptKey] < DEBOUNCE_TIME then
            continue
        end
        
        -- ✅ ATIVA!
        obj:InputHoldBegin()
        wait(0.05)
        obj:InputHoldEnd()
        
        triggeredPrompts[promptKey] = tick()
        print("🎯 AUTO → " .. obj.ActionText .. " (" .. math.floor(distance) .. " studs)")
    end
end

-- LOOP PRINCIPAL
RunService.Heartbeat:Connect(function()
    local now = tick()
    if now - lastUpdate < UPDATE_INTERVAL then return end
    lastUpdate = now
    
    findAndTriggerPrompts()
end)

-- RESET SPAWN
player.CharacterAdded:Connect(function(newChar)
    character = newChar
    rootPart = newChar:WaitForChild("HumanoidRootPart")
    triggeredPrompts = {}
    print("👤 AutoPrompt resetado")
end)

print("🎉 AUTO PROXIMITY SEMPRE ATIVO! 🚀 (15 studs)")

    -- LocalScript: TELEPORTE NPC MAIS PRÓXIMO (INSTANTÂNEO SEMPRE) - até 500 studs

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local rootPart = character:WaitForChild("HumanoidRootPart")
local TARGETS_FOLDER = workspace.Worlds:WaitForChild("Targets")
local CLIENTS_FOLDER = TARGETS_FOLDER:WaitForChild("Clients")

print("✅ Teleporte INSTANTÂNEO SEMPRE ATIVO! (até 500 studs)")

-- 🔥 CONFIGS
local MAX_DISTANCE = 500
local UPDATE_INTERVAL = 0.1     -- checa a cada \~0.1s
local MIN_DISTANCE_TO_TP = 0.1  -- ignora se já estiver muito perto
local DEBOUNCE_TP = 0.25        -- evita spamar tp no mesmo alvo

-- Controle
local lastUpdate = 0
local lastTarget = nil
local lastTpTime = 0
local tpCount = 0

-- 🔥 FUNÇÃO NPC MAIS PRÓXIMO
local function getClosestEnemy()
    local closest = nil
    local smallestDist = math.huge

    for _, model in ipairs(CLIENTS_FOLDER:GetChildren()) do
        if not model:IsA("Model") then continue end
        
        local humanoid = model:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 or humanoid:GetState() == Enum.HumanoidStateType.Dead then 
            continue 
        end
        
        local hrp = model:FindFirstChild("HumanoidRootPart") 
                 or model:FindFirstChild("Torso") 
                 or model:FindFirstChild("UpperTorso")
                 or model.PrimaryPart
        
        if not hrp or hrp.Position.Y < -500 then continue end

        local dist = (hrp.Position - rootPart.Position).Magnitude
        if dist < smallestDist then
            smallestDist = dist
            closest = hrp
        end
    end

    return closest, smallestDist
end

-- 🔥 TP INSTANTÂNEO (sempre)
local function instantTP(targetPart, distance)
    local offset = (rootPart.Position - targetPart.Position).Unit * 4
    local desiredPos = targetPart.Position + offset
    local goalCFrame = CFrame.new(desiredPos, targetPart.Position)
    
    rootPart.CFrame = goalCFrame
    
    print("⚡ TP INSTANTÂNEO → " .. math.floor(distance) .. " studs")
end

-- 🔥 LOOP PRINCIPAL (sempre ativo)
RunService.Heartbeat:Connect(function()
    local now = tick()
    if now - lastUpdate < UPDATE_INTERVAL then return end
    lastUpdate = now
    
    local closestPart, distance = getClosestEnemy()
    
    if not closestPart or distance > MAX_DISTANCE then 
        lastTarget = nil
        return 
    end
    
    -- Evita ficar trocando pro mesmo alvo sem necessidade
    if closestPart == lastTarget and now - lastTpTime < DEBOUNCE_TP then
        return
    end
    
    if distance < MIN_DISTANCE_TO_TP then return end
    
    lastTarget = closestPart
    lastTpTime = now
    tpCount = tpCount + 1
    
    print(string.format("🎯 TP #%d → %d studs", tpCount, math.floor(distance)))
    instantTP(closestPart, distance)
end)

-- RESET AO MORRER / RESPAWN
player.CharacterAdded:Connect(function(newChar)
    character = newChar
    rootPart = newChar:WaitForChild("HumanoidRootPart")
    lastTarget = nil
    print("👤 Personagem respawnado → TP resetado")
end)

print("🎉 TELEPORTE 100% INSTANTÂNEO ATIVADO! (0–500 studs) 🚀")
