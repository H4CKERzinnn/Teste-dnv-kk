-- Auto Raid (6min Easy) ↔ Dungeon (6min Easy) INFINITO
-- Teleporte de saída: novo remote pro "Lobby"
-- Auto TP: versão simples sempre ativo (primeira que você mandou, sem toggle/UI)

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players           = game:GetService("Players")
local RunService        = game:GetService("RunService")

local Remote = ReplicatedStorage:WaitForChild("ffrostflame_bridgenet2@1.0.0"):WaitForChild("dataRemoteEvent")

-- === CONFIGS ===
local RAID_PLACE_ID     = 10301056827      -- Raid
local DUNGEON_PLACE_ID  = 4286839868      -- Dungeon Easy
local RAID_MAP          = "TitanTown"
local DUNGEON_MAP       = "CrystalCave"
local RAID_DIFFICULTY   = "Easy"
local DUNGEON_DIFFICULTY = "Easy"
local SLOTS             = 6

-- Tempos (6 minutos cada + safety)
local MODE_TIME         = 6 * 60 + 20     -- 6min + 20s buffer
local TELEPORT_DELAY    = 3               -- delay após sair
local CREATE_DELAY      = 3 + math.random() * 2  -- random pra parecer mais natural

-- Pasta dos inimigos (do seu primeiro TP)
local ENEMIES_FOLDER = workspace:WaitForChild("_ENEMIES"):WaitForChild("Client")

-- Variáveis do auto TP
local player    = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local rootPart  = character:WaitForChild("HumanoidRootPart")
local lastUpdate = 0
local UPDATE_INTERVAL_TP = 0.5      -- checa a cada 0.5s
local MIN_DISTANCE_TO_TP = 14
local OFFSET_STUDS       = 4

-- =================================================
-- AUTO TP SIMPLES (sempre ativo - primeira versão)
-- =================================================
local function getClosestEnemy()
    local closest = nil
    local smallestDist = math.huge

    for _, model in ipairs(ENEMIES_FOLDER:GetChildren()) do
        if not model:IsA("Model") then continue end
        
        local humanoid = model:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then continue end
        
        local targetPart = model:FindFirstChild("HumanoidRootPart") 
                         or model:FindFirstChild("UpperTorso")
                         or model:FindFirstChild("Torso") 
                         or model.PrimaryPart
                         or model:FindFirstChildWhichIsA("BasePart")
        
        if not targetPart then continue end
        
        local dist = (targetPart.Position - rootPart.Position).Magnitude
        if dist < smallestDist then
            smallestDist = dist
            closest = targetPart
        end
    end

    return closest, smallestDist
end

local function instantTP(targetPart)
    if not targetPart then return end
    
    local direction = (rootPart.Position - targetPart.Position).Unit
    local desiredPos = targetPart.Position + (direction * OFFSET_STUDS)
    
    rootPart.CFrame = CFrame.new(desiredPos, targetPart.Position)
end

-- Loop do TP (sempre rodando)
RunService.Heartbeat:Connect(function()
    if tick() - lastUpdate < UPDATE_INTERVAL_TP then return end
    lastUpdate = tick()
    
    local closestPart, distance = getClosestEnemy()
    
    if not closestPart then return end
    if distance < MIN_DISTANCE_TO_TP then return end
    
    instantTP(closestPart)
end)

-- Reset TP ao respawn
player.CharacterAdded:Connect(function(newChar)
    character = newChar
    rootPart = newChar:WaitForChild("HumanoidRootPart")
    lastUpdate = tick() - UPDATE_INTERVAL_TP + 0.5
end)

-- =================================================
-- FUNÇÕES RAID / DUNGEON / SAÍDA
-- =================================================
local function createRaid()
    local args = { {{"GamemodeSystem", "Create", "Raid", RAID_MAP, RAID_DIFFICULTY, n = SLOTS}, "\002"} }
    pcall(function() Remote:FireServer(unpack(args)) end)
    print("🗡️ CRIANDO RAID Easy...")
end

local function startRaid()
    local args = { {{"GamemodeSystem", "Start", "Raid", RAID_PLACE_ID, n = 4}, "\002"} }
    pcall(function() Remote:FireServer(unpack(args)) end)
    print("🚀 RAID Easy INICIADA!")
end

local function createDungeon()
    local args = { {{"GamemodeSystem", "Create", "Dungeon", DUNGEON_MAP, DUNGEON_DIFFICULTY, n = SLOTS}, "\002"} }
    pcall(function() Remote:FireServer(unpack(args)) end)
    print("💎 CRIANDO DUNGEON Easy...")
end

local function startDungeon()
    local args = { {{"GamemodeSystem", "Start", "Dungeon", DUNGEON_PLACE_ID, n = 4}, "\002"} }
    pcall(function() Remote:FireServer(unpack(args)) end)
    print("✨ DUNGEON Easy INICIADA!")
end

local function exitToLobby()
    local args = {
        {
            {
                "TeleportSystem",
                "To",
                "Lobby",
                n = 3
            },
            "\002"
        }
    }
    pcall(function() Remote:FireServer(unpack(args)) end)
    print("🏠 TELEPORTANDO PARA O LOBBY...")
end

-- =================================================
-- LOOP PRINCIPAL (Raid ↔ Dungeon infinito)
-- =================================================
print("🔄 Auto Raid/Dungeon Easy 6min INFINITO iniciado!")
print("Auto TP sempre ativo (inimigos em _ENEMIES.Client)")
print("Saída: novo remote → Lobby")

local isDoingRaid = true

while true do
    if isDoingRaid then
        -- RAID Easy 6min
        print("\n=== RAID EASY (6min) ===")
        createRaid()
        task.wait(CREATE_DELAY)
        startRaid()
        print("⏳ Aguardando 6min na Raid...")
        task.wait(MODE_TIME)
        
        print("🏃 Saindo da Raid...")
        exitToLobby()
        task.wait(TELEPORT_DELAY)
        
    else
        -- DUNGEON Easy 6min
        print("\n=== DUNGEON EASY (6min) ===")
        createDungeon()
        task.wait(CREATE_DELAY)
        startDungeon()
        print("⏳ Aguardando 6min na Dungeon...")
        task.wait(MODE_TIME)
        
        print("🏃 Saindo da Dungeon...")
        exitToLobby()
        task.wait(TELEPORT_DELAY)
    end
    
    isDoingRaid = not isDoingRaid
    print("🔄 Alternando para: " .. (isDoingRaid and "RAID Easy" or "DUNGEON Easy"))
    task.wait(2)  -- safety extra entre ciclos
end
